# Enable Root in VPS
```
sudo sed -i 's/#PermitRootLogin prohibit-password/PermitRootLogin yes/' /etc/ssh/sshd_config && sudo systemctl restart ssh && sudo passwd
```
##

# change ssh port
```
nano /etc/ssh/sshd_config
```
```
systemctl restart sshd
```
